\begin{frame}{Fine tuning the reward}
    \begin{itemize}
        \item - \textbf{Final Reward :} total profit over a given period
        \item - \textbf{Portfolio Change :} difference between the portfolio value before and after an action is taken
        \item - \textbf{Penalty Loss :} negative reward whenever the agent make a losing trade
        \item - \textbf{Portfolio Delta Risk :} risk-reward based on the volatility associated to an action
    \end{itemize}
\end{frame}
